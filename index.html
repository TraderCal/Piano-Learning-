let selectedRoot = 0;

// ── PIANO LAYOUT ─────────────────────────────────────────────────────────────

// 2 octaves: C to B (keys C D E F G A B C D E F G A B)

const WHITE_NOTES = ['C','D','E','F','G','A','B','C','D','E','F','G','A','B'];

const BLACK_POSITIONS = [1,3,null,6,8,10,null,13,15,null,18,20,22,null]; // semitone positions


// Build piano

function buildPiano(litSemitones) {

  // We'll render 2 octaves, keys 0-27 semitones (0=C, 27=B+1oct)

  // White keys: C D E F G A B | C D E F G A B

  const whiteNotes = ['C','D','E','F','G','A','B','C','D','E','F','G','A','B'];

  // Black keys between whites (relative): after C,D / after F,G,A / (repeat)

  // Position (px offset from start of each white key): 

  const blackAfterWhite = [0,1,3,5,6,8,10, 0+7, 1+7, 3+7, 5+7, 6+7, 8+7, 10+7];

  // whiteKey semitones

  const whiteSemi = [0,2,4,5,7,9,11, 12,14,16,17,19,21,23];

  const blackSemi =  [1,3,6,8,10, 13,15,18,20,22]; // all sharps in 2 octaves

  const blackAfterIdx = [0,1,3,5,6, 7,8,10,12,13]; // which white key index they go after


  const wKeyW = 36, wKeyH = 110, bKeyW = 24, bKeyH = 68;

  const totalW = whiteNotes.length * wKeyW + 20;


  let svgW = totalW, svgH = wKeyH + 20;


  // Build HTML

  let pianoHtml = `<div class="piano" style="width:${totalW}px;height:${wKeyH+20}px;position:relative;">`;


  // White keys

  whiteNotes.forEach((note, i) => {

    const semi = whiteSemi[i];

    const lit = litSemitones.includes(semi % 12);

    const isRoot = (semi % 12) === 0; // root semitone = 0 relative

    pianoHtml += `<div class="white-key${lit ? (isRoot ? ' root-lit' : ' lit') : ''}" style="position:absolute;left:${10+i*wKeyW}px;top:10px;">

      <span class="key-note">${note}</span>

    </div>`;

  });


  // Black keys

  blackAfterIdx.forEach((whiteIdx, i) => {

    const semi = blackSemi[i];

    const lit = litSemitones.includes(semi % 12);

    const isRoot = (semi % 12) === 0;

    const leftPos = 10 + whiteIdx * wKeyW + wKeyW - bKeyW/2;

    const chromNote = CHROMATIC[semi % 12];

    const enharmonic = ENHARMONIC[chromNote];

    const label = enharmonic ? chromNote + '/' + enharmonic : chromNote;

    pianoHtml += `<div class="black-key${lit ? (isRoot ? ' root-lit' : ' lit') : ''}" style="left:${leftPos}px;top:10px;">

      <span class="key-note">${chromNote.replace('#','♯')}</span>

    </div>`;

  });


  pianoHtml += `</div>`;

  return pianoHtml;

}


// ── NOTE DISPLAY ─────────────────────────────────────────────────────────────

function getNoteAtSemitone(rootIdx, semitoneOffset) {

  const idx = (rootIdx + semitoneOffset) % 12;

  const note = CHROMATIC[idx];

  // Choose sharps or flats based on root

  const flatRoots = [1,3,6,8,10];

  if (flatRoots.includes(rootIdx) && ENHARMONIC[note]) {

    return ENHARMONIC[note];

  }

  return note;

}


function renderDetail(chord) {

  if (!chord) {

    return `<div class="empty-state">

      <div class="big">🎹</div>

      <p>Select a chord above to see it on the piano and learn how it's built.</p>

    </div>`;

  }


  const cat = CATEGORIES.find(c => c.id === chord.cat);

  const accent = cat ? cat.accent : '#e8c87a';


  // Build semitone set (relative, 0-11)

  const relativeSemitones = chord.intervals.map(iv => INTERVAL_MAP[iv] % 12);


  // Get actual notes

  const noteNames = chord.intervals.map(iv => {

    const semi = INTERVAL_MAP[iv];

    return getNoteAtSemitone(selectedRoot, semi);

  });


  // Build symbol with root

  const rootDisplay = ROOTS[selectedRoot].replace('/', '/');

  const chordSymbol = chord.symbol.replace(/^C/, rootDisplay.split('/')[0]);


  // Lit semitones for piano (shifted by root)

  // Piano shows 2 octaves starting at C. We need to show correct positions.

  // Map: for each chord interval, find which piano semitone it is

  // Piano semitone 0 = first C on keyboard. Root might not be C.

  // We light: (rootSemitone + intervalSemitone) % 12 for each interval

  const pianoSemitones = chord.intervals.map(iv => {

    return (ROOT_SEMITONES[selectedRoot] + INTERVAL_MAP[iv]) % 12;

  });


  return `

    <div class="detail-panel" style="--accent:${accent}">

      <div class="detail-header">

        <div>

          <div class="detail-title">${rootDisplay} ${chord.name}</div>

          <div class="detail-symbol">${chordSymbol} &nbsp;·&nbsp; ${chord.intervals.join(' - ')}</div>

        </div>

        <div class="formula-badge">

          <span>Intervals</span>

          <strong>${chord.intervals.join('  ')}</strong>

        </div>

      </div>


      <!-- Notes row -->

      <div class="notes-row">

        ${chord.intervals.map((iv, i) => `

          <div class="note-pill" style="animation-delay:${i*0.05}s">

            <div class="note-dot${iv==='R'?' root-dot':''}"></div>

            <span class="note-name">${noteNames[i].replace('#','♯')}</span>

            <span class="note-role">${INTERVAL_NAMES[iv]}</span>

          </div>

        `).join('')}

      </div>


      <!-- Piano -->

      <div class="piano-wrap">

        <div class="piano-label">Keys to press (highlighted)</div>

        ${buildPiano(pianoSemitones)}

      </div>


      <!-- Info grid -->

      <div class="section-grid">

        <div class="info-box">

          <h4>🎧 How it sounds</h4>

          <p>${chord.sound}</p>

        </div>

        <div class="info-box">

          <h4>🎵 Musical context</h4>

          <p>${chord.vibe}</p>

        </div>

        <div class="tip-box">

          <span class="tip-icon">💡</span>

          <p><strong>Theory tip:</strong> ${chord.tip}</p>

        </div>

        ${chord.omit ? `<div class="info-box">

          <h4>✂️ Voicing tip</h4>

        </div>` : ''}

        <div class="memory-trick">

          <p>${chord.memory}</p>

        </div>

      </div>

    </div>

  `;

}

function render() {

  // Root bar

  const rootBar = document.getElementById('rootBar');

  rootBar.innerHTML = ROOTS.map((r, i) =>

    `<button class="root-btn${selectedRoot===i?' active':''}" onclick="selectRoot(${i})">${r}</button>`

  ).join('');


  // Category tabs

  const catTabs = document.getElementById('catTabs');

  catTabs.innerHTML = CATEGORIES.map(c =>

    `<button class="cat-tab${selectedCat===c.id?' active':''}" onclick="selectCat('${c.id}')" style="${selectedCat===c.id?`color:${c.accent}`:''}">

      ${c.label}

    </button>`

  ).join('');


  // Chord grid

  const filtered = CHORDS.filter(ch => ch.cat === selectedCat);

  const cat = CATEGORIES.find(c => c.id === selectedCat);

  const accent = cat ? cat.accent : '#e8c87a';


  const chordGrid = document.getElementById('chordGrid');

  chordGrid.innerHTML = filtered.map(ch =>

    `<div class="chord-card${selectedChord && selectedChord.id===ch.id?' selected':''}"

         style="--accent:${accent}"

         onclick="selectChord('${ch.id}')">

      <div class="chord-name">${ch.name}</div>

      <div class="chord-symbol">${ch.symbol.replace(/^C/, ROOTS[selectedRoot].split('/')[0])}</div>

      <div class="chord-tagline">${ch.intervals.join(' – ')}</div>

    </div>`

  ).join('');


  // Detail

  document.getElementById('detailPanel').innerHTML = renderDetail(selectedChord);

}


function selectRoot(i) { selectedRoot = i; render(); }

function selectCat(id) {

  selectedCat = id;

  selectedChord = null;

  render();

}

function selectChord(id) {

  selectedChord = CHORDS.find(c => c.id === id) || null;

  render();

  setTimeout(() => {

    const panel = document.getElementById('detailPanel');

    if (panel) panel.scrollIntoView({ behavior: 'smooth', block: 'nearest' });

  }, 50);

}
